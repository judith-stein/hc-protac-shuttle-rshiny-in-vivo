# Modeling assignemnts -----------------------------------

 SF ~ 10^9 * 1 / 6.023 * 10^-23

 # Add total tumor volume in mmÂ³
 V_tumor_mm3 <- V_tumor_pro_mm3 + V_tumor_dyi_1_mm3 + V_tumor_dyi_2_mm3 + V_tumor_dyi_3_mm3

 # Tumor volume in liter
 V_tumor ~ V_tumor_mm3 * 10^-6

 # Number of tumor cells 
 NC_Tumor ~ NCL_tumor * V_tumor
 
 # Radius tumor in cm
 R_Tumor ~ (3 * V_tumor / (4 * pi))^(1/3) * 10

 # Total antibody concentration in plasma
 Ab_C1_t_nM <- (Ab_C1_f + Ab_C1_b1 + Ab_C1_m1 + Ab_C1_b2 + Ab_C1_m2 + Ab_C1_b3 + Ab_C1_m3 + Ab_C1_b4 + Ab_C1_m4 + Ab_C1_b1_m1 + Ab_C1_b1_m2 + Ab_C1_b1_m3 + Ab_C1_b2_m1 + Ab_C1_b2_m2 + Ab_C1_b3_m1) / V_C1_Ab
 ngmL_Ab_C1_t <- Ab_C1_t_nM * 1e-03 * MW_Ab

 # Total antibody concentration in peripheral compartment
 Ab_C2_t_nM <- (Ab_C2_f + Ab_C2_b1 + Ab_C2_m1 + Ab_C2_b2 + Ab_C2_m2 + Ab_C2_b3 + Ab_C2_m3 + Ab_C2_b4 + Ab_C2_m4 + Ab_C2_b1_m1 + Ab_C2_b1_m2 + Ab_C2_b1_m3 + Ab_C2_b2_m1 + Ab_C2_b2_m2 + Ab_C2_b3_m1) / V_C2_Ab

  # Total antibody concentration in tumor
 Ab_ex_t <- Ab_ex_f + Ab_ex_b1 + Ab_ex_m1 + Ab_ex_b2 + Ab_ex_m2 + Ab_ex_b3 + Ab_ex_m3 + Ab_ex_b4 + Ab_ex_m4 + Ab_ex_b1_m1 + Ab_ex_b1_m2 + Ab_ex_b1_m3 + Ab_ex_b2_m1 + Ab_ex_b2_m2 + Ab_ex_b3_m1 
           + (Ab_cell_f_b_ag + Ab_cell_b1_b_ag + Ab_cell_m1_b_ag + Ab_cell_b2_b_ag + Ab_cell_m2_b_ag + Ab_cell_b3_b_ag + Ab_cell_m3_b_ag + Ab_cell_b4_b_ag + Ab_cell_m4_b_ag + Ab_cell_b1_m1_b_ag + Ab_cell_b1_m2_b_ag + Ab_cell_b1_m3_b_ag + Ab_cell_b2_m1_b_ag + Ab_cell_b2_m2_b_ag +  Ab_cell_b3_m1_b_ag) * NC_Tumor * SF / V_tumor
           # no Ab_cell_lyso_i, since no degradation in lysosome of antibody
 ngmL_Ab_ex_t <- Ab_ex_t * 1e-03 * MW_Ab

 # Total Protac in plasma
 Drug_C1_t <- Drug_C1_f / SF * V_C1_Drug * BW + Drug_C1_b_ntp / SF * V_C1_Drug * BW + (Ab_C1_b1 * 1 + Ab_C1_b2 * 2 + Ab_C1_b3 * 3 + Ab_C1_b4 * 4) / SF * BW + (Ab_C1_b1_m1 + Ab_C1_b1_m2 + Ab_C1_b1_m3) * 1 / SF * BW + (Ab_C1_b2_m1 + Ab_C1_b2_m2) * 2 / SF * BW + Ab_C1_b3_m1 * 3 / SF * BW
 ngmL_Drug_C1_t <- Drug_C1_t * MW_Drug * SF / V_C1_Drug / BW * 1e-03

  # Total Protac in tumor
 Drug_ex_t <- Drug_ex_f / SF + (Ab_ex_b1 * 1 + Ab_ex_b2 * 2 + Ab_ex_b3 * 3 + Ab_ex_b4 * 4) / SF * V_tumor + (Ab_ex_b1_m1 + Ab_ex_b1_m2 + Ab_ex_b1_m3) * 1 / SF * V_tumor + (Ab_ex_b2_m1 + Ab_ex_b2_m2) * 2 / SF * V_tumor + Ab_ex_b3_m1 * 3 / SF * V_tumor
              + (Ab_cell_b1_b_ag * 1 + Ab_cell_b2_b_ag * 2 + Ab_cell_b3_b_ag * 3 + Ab_cell_b4_b_ag * 4 + Ab_cell_b1_m1_b_ag + Ab_cell_b1_m2_b_ag + Ab_cell_b1_m3_b_ag + (Ab_cell_b2_m1_b_ag + Ab_cell_b2_m2_b_ag) * 2 +  Ab_cell_b3_m1_b_ag * 3) * NC_Tumor
              + (Drug_cell_lyso_f + Drug_cell_cyto_f + Drug_cell_cyto_b_dt) * NC_Tumor
 ngmL_Drug_ex_t <- Drug_ex_t * MW_Drug * SF / V_tumor * 1e-03

 # Total ADC amount in plasma
 ADC_C1_t <- (ADC_C1_f + ADC_C1_b1 + ADC_C1_b2 + ADC_C1_m1 + ADC_C1_m2 + ADC_C1_b1_m1)

 # Mean DAR in plasma
 # only Protacs
 DAR1 <- ((Ab_C1_f * 0 + (Ab_C1_b1 * 1 + Ab_C1_b2 * 2 + Ab_C1_b3 * 3 + Ab_C1_b4 * 4) + (Ab_C1_b1_m1 + Ab_C1_b1_m2 + Ab_C1_b1_m3) * 1 + (Ab_C1_b2_m1 + Ab_C1_b2_m2) * 2 + Ab_C1_b3_m1 * 3)) / (Ab_C1_t_nM * V_C1_Ab + ADC_C1_t)
         + ((ADC_C1_t - ADC_C1_m1 - ADC_C1_m2) / ADC_C1_t + DAR) * ADC_C1_t / (Ab_C1_t_nM * V_C1_Ab + ADC_C1_t)

# Protacs + Metabolite1
 DAR2 <- ((Ab_C1_f * 0 + (Ab_C1_b1 * 1 + Ab_C1_b2 * 2 + Ab_C1_b3 * 3 + Ab_C1_b4 * 4) + (Ab_C1_m1 * 1 + Ab_C1_m2 * 2 + Ab_C1_m3 * 3 + Ab_C1_m4 * 4) + (Ab_C1_b1_m1 * 2 + Ab_C1_b1_m2 * 3 + Ab_C1_b1_m3 * 4) + (Ab_C1_b2_m1 * 2 + Ab_C1_b2_m2 * 3) + Ab_C1_b3_m1 * 4)) / (Ab_C1_t_nM * V_C1_Ab)
 

 # DAR for ADC
 d/dt(DAR) <- - K_ADC_dec * DAR


 # Ordinary differential equations ------------------------------
 ########################

 # Amount (nmol/kg) of free ADC (bound to 0 Protacs) in central compartment/plasma
 d/dt(ADC_C1_f) <- - (CL_Ab / V_C1_Ab) * ADC_C1_f 
                   - (CLD_Ab / V_C1_Ab) * ADC_C1_f 
                   + (CLD_Ab / V_C2_Ab) * ADC_C2_f 
                   - (ADC_C1_f / V_C1_Ab - ADC_ex_f / E_Ab) * V_tumor / BW * (2 * P_Ab * R_Cap / R_Krogh^2 + 6 * D_Ab / R_Tumor^2)
                   - K_Ab_Drug_on * 2 * ADC_C1_f * Drug_C1_f
                   + K_Ab_Drug_off * ADC_C1_b1
                   - K_Ab_Meta_on * 2 * ADC_C1_f * Meta1_C1_f
                   + K_Ab_Meta_off * ADC_C1_m1

 # Amount (nmol/kg) of ADC bound to 1 Protac in central compartment/plasma
 d/dt(ADC_C1_b1) <- - (CL_Ab / V_C1_Ab) * ADC_C1_b1 
                   - (CLD_Ab / V_C1_Ab) * ADC_C1_b1 
                   + (CLD_Ab / V_C2_Ab) * ADC_C2_b1 
                   - (ADC_C1_b1 / V_C1_Ab - ADC_ex_b1 / E_Ab) * V_tumor / BW * (2 * P_Ab * R_Cap / R_Krogh^2 + 6 * D_Ab / R_Tumor^2)
                   + K_Ab_Drug_on * 2 * ADC_C1_f * Drug_C1_f
                   - K_Ab_Drug_off * ADC_C1_b1
                   - K_Ab_Drug_on * ADC_C1_b1 * Drug_C1_f
                   + K_Ab_Drug_off * ADC_C1_b2
                   - K_Ab_Meta_on * ADC_C1_b1 * Meta1_C1_f
                   + K_Ab_Meta_off * ADC_C1_b1_m1

 # Amount (nmol/kg) of ADC bound to 2 Protacs in central compartment/plasma
 d/dt(ADC_C1_b2) <- - (CL_Ab / V_C1_Ab) * ADC_C1_b2 
                   - (CLD_Ab / V_C1_Ab) * ADC_C1_b2 
                   + (CLD_Ab / V_C2_Ab) * ADC_C2_b2 
                   - (ADC_C1_b2 / V_C1_Ab - ADC_ex_b2 / E_Ab) * V_tumor / BW * (2 * P_Ab * R_Cap / R_Krogh^2 + 6 * D_Ab / R_Tumor^2)
                   + K_Ab_Drug_on * ADC_C1_b1 * Drug_C1_f
                   - K_Ab_Drug_off * ADC_C1_b2

 # Amount (nmol/kg) of ADC bound to 1 Metabolite1 in central compartment/plasma
 d/dt(ADC_C1_m1) <- - (CL_Ab / V_C1_Ab) * ADC_C1_m1 
                   - (CLD_Ab / V_C1_Ab) * ADC_C1_m1 
                   + (CLD_Ab / V_C2_Ab) * ADC_C2_m1 
                   - (ADC_C1_m1 / V_C1_Ab - ADC_ex_m1 / E_Ab) * V_tumor / BW * (2 * P_Ab * R_Cap / R_Krogh^2 + 6 * D_Ab / R_Tumor^2)
                   + K_Ab_Meta_on * 2 * ADC_C1_f * Meta1_C1_f
                   - K_Ab_Meta_off * ADC_C1_m1
                   - K_Ab_Drug_on * ADC_C1_m1 * Drug_C1_f
                   + K_Ab_Drug_off * ADC_C1_b1_m1
                   - K_Ab_Meta_on * ADC_C1_m1 * Meta1_C1_f
                   + K_Ab_Meta_off * ADC_C1_m2

 # Amount (nmol/kg) of ADC bound to 2 Metabolite1 in central compartment/plasma
 d/dt(ADC_C1_m2) <- - (CL_Ab / V_C1_Ab) * ADC_C1_m2 
                   - (CLD_Ab / V_C1_Ab) * ADC_C1_m2 
                   + (CLD_Ab / V_C2_Ab) * ADC_C2_m2 
                   - (ADC_C1_m2 / V_C1_Ab - ADC_ex_m2 / E_Ab) * V_tumor / BW * (2 * P_Ab * R_Cap / R_Krogh^2 + 6 * D_Ab / R_Tumor^2)
                   + K_Ab_Meta_on * ADC_C1_m1 * Meta1_C1_f
                   - K_Ab_Meta_off * ADC_C1_m2

 # Amount (nmol/kg) of ADC bound to 1 Protac and 1 Metabolite1 in central compartment/plasma
 d/dt(ADC_C1_b1_m1) <- - (CL_Ab / V_C1_Ab) * ADC_C1_b1_m1
                   - (CLD_Ab / V_C1_Ab) * ADC_C1_b1_m1 
                   + (CLD_Ab / V_C2_Ab) * ADC_C2_b1_m1
                   - (ADC_C1_b1_m1 / V_C1_Ab - ADC_ex_b1_m1 / E_Ab) * V_tumor / BW * (2 * P_Ab * R_Cap / R_Krogh^2 + 6 * D_Ab / R_Tumor^2)
                   + K_Ab_Meta_on * ADC_C1_b1 * Meta1_C1_f
                   - K_Ab_Meta_off * ADC_C1_b1_m1
                   + K_Ab_Drug_on * ADC_C1_m1 * Drug_C1_f
                   - K_Ab_Drug_off * ADC_C1_b1_m1
 
 # Amount (nmol/kg) of free Antibody (bound to 0 Protacs) in central compartment/plasma
 d/dt(Ab_C1_f) <- - (CL_Ab / V_C1_Ab) * Ab_C1_f 
                   - (CLD_Ab / V_C1_Ab) * Ab_C1_f 
                   + (CLD_Ab / V_C2_Ab) * Ab_C2_f 
                   - (Ab_C1_f / V_C1_Ab - Ab_ex_f / E_Ab) * V_tumor / BW * (2 * P_Ab * R_Cap / R_Krogh^2 + 6 * D_Ab / R_Tumor^2)
                   - K_Ab_Drug_on * max * Ab_C1_f * Drug_C1_f
                   + K_Ab_Drug_off * Ab_C1_b1
                   - K_Ab_Meta_on * max * Ab_C1_f * Meta1_C1_f
                   + K_Ab_Meta_off * Ab_C1_m1
 
 # Amount (nmol/kg) of Antibody bound to 1 Protac in central compartment/plasma
 d/dt(Ab_C1_b1) <- - (CL_Ab / V_C1_Ab) * Ab_C1_b1 
                   - (CLD_Ab / V_C1_Ab) * Ab_C1_b1 
                   + (CLD_Ab / V_C2_Ab) * Ab_C2_b1 
                   - (Ab_C1_b1 / V_C1_Ab - Ab_ex_b1 / E_Ab) * V_tumor / BW * (2 * P_Ab * R_Cap / R_Krogh^2 + 6 * D_Ab / R_Tumor^2)
                   + K_Ab_Drug_on * max * Ab_C1_f * Drug_C1_f
                   - K_Ab_Drug_off * Ab_C1_b1
                   - K_Ab_Drug_on * (max - 1) * Ab_C1_b1 * Drug_C1_f + K_Ab_Drug_off * Ab_C1_b2
                              - K_Ab_Meta_on * (max - 1) * Ab_C1_b1 * Meta1_C1_f

d/dt(Ab_C1_b2) <- - (CL_Ab / V_C1_Ab) * Ab_C1_b2

                  + (CLD_Ab / V_C2_Ab) * Ab_C2_b2 
                  - (Ab_C1_b2 / V_C1_Ab - Ab_ex_b2 / E_Ab) * V_tumor / BW * (2 * P_Ab * R_Cap / R_Krogh^2 + 6 * D_Ab / R_Tumor^2)
                  + K_Ab_Drug_on * (max - 1) * Ab_C1_b1 * Drug_C1_f
                  - K_Ab_Drug_off * Ab_C1_b2
                  - K_Ab_Drug_on * (max - 2) * Ab_C1_b2 * Drug_C1_f + K_Ab_Drug_off * Ab_C1_b3
                                - K_Ab_Meta_on * (max - 2) * Ab_C1_b2 * Meta1_C1_f

# Number of Antibody molecules bound to 1 Protac and 3 Metabolites1 in lysosomal space on a single tumor cell
d/dt(Ab_cell_lyso_b1_m3) <- K_Ab_cell_int * Ab_cell_b1_m3_b_ag

                                          + K_Ab_cell_lyso_pino * Ab_ex_b1_m3 / (E_Ab * SF)

# Number of Antibody molecules bound to 2 Protacs and 1 Metabolites1 in lysosomal space on a single tumor cell
d/dt(Ab_cell_lyso_b2_m1) <- K_Ab_cell_int * Ab_cell_b2_m1_b_ag 
                                          - K_Ab_deg * Ab_cell_lyso_b2_m1 
                                          + K_Ab_cell_lyso_pino * Ab_ex_b2_m1 / (E_Ab * SF) 
                                          - log(2) / DT_tumor * Ab_cell_lyso_b2_m1
                                          
# Number of Antibody molecules bound to 2 Protacs and 2 Metabolites1 in lysosomal space on a single tumor cell
d/dt(Ab_cell_lyso_b2_m2) <- K_Ab_cell_int * Ab_cell_b2_m2_b_ag 
                                          - K_Ab_deg * Ab_cell_lyso_b2_m2 
                                          + K_Ab_cell_lyso_pino * Ab_ex_b2_m2 / (E_Ab * SF) 
                                          - log(2) / DT_tumor * Ab_cell_lyso_b2_m2
                                          
# Number of Antibody molecules bound to 3 Protacs and 1 Metabolite1 in lysosomal space on a single tumor cell
d/dt(Ab_cell_lyso_b3_m1) <- K_Ab_cell_int * Ab_cell_b3_m1_b_ag 
                            - K_Ab_deg * Ab_cell_lyso_b3_m1 
                            + K_Ab_cell_lyso_pino * Ab_ex_b3_m1 / (E_Ab * SF) 
                            - log(2) / DT_tumor * Ab_cell_lyso_b3_m1                              
          
                            

 # Number of unbound (free) drug molecules in lysosomal space on a single tumor cell
 d/dt(Drug_cell_lyso_f) <- - K_Drug_lyso_out * (V_cell / V_cell_lyso) * Drug_cell_lyso_f 
                           + K_Drug_lyso_in * Drug_cell_cyto_f 
                           - log(2) / DT_tumor * Drug_cell_lyso_f
                           - K_Drug_cell_met * Drug_cell_lyso_f
                           + K_Ab_deg * (ADC_cell_lyso_b1 + ADC_cell_lyso_b2 * 2 + ADC_cell_lyso_b1_m1)
                           + K_Ab_deg * (ADC_cell_lyso_b1 + ADC_cell_lyso_b2 + ADC_cell_lyso_m1 + ADC_cell_lyso_m2 + ADC_cell_lyso_b1_m1) * DAR
                           + K_Ab_deg * ((Ab_cell_lyso_b1 * 1 + Ab_cell_lyso_b2 * 2 + Ab_cell_lyso_b3 * 3 + Ab_cell_lyso_b4 * 4) + (Ab_cell_b1_m1_b_ag + Ab_cell_b1_m2_b_ag + Ab_cell_b1_m3_b_ag) + (Ab_cell_b2_m1_b_ag + Ab_cell_b2_m2_b_ag) * 2 +  Ab_cell_b3_m1_b_ag * 3)

 # Number of unbound (free) Metabolite1 molecules in lysosomal space on a single tumor cell
 d/dt(Meta1_cell_lyso_f) <- - K_Meta_lyso_out * (V_cell / V_cell_lyso) * Meta1_cell_lyso_f 
                             + K_Meta_lyso_in * Meta1_cell_cyto_f 
                             - log(2) / DT_tumor * Meta1_cell_lyso_f
                             + K_Drug_cell_met * Drug_cell_lyso_f
                             + K_Ab_deg * (ADC_cell_lyso_m1 + ADC_cell_lyso_m2 * 2 + ADC_cell_lyso_b1_m1)
                             + K_Ab_deg * ((Ab_cell_lyso_m1 * 1 + Ab_cell_lyso_m2 * 2 + Ab_cell_lyso_m3 * 3 + Ab_cell_lyso_m4 * 4) + (Ab_cell_lyso_b1_m1 * 1 + Ab_cell_lyso_b1_m2 * 2 + Ab_cell_lyso_b1_m3 * 3) + (Ab_cell_lyso_b2_m1 * 1 + Ab_cell_lyso_b2_m2 * 2) +  Ab_cell_b3_m1_b_ag)

 # Number of unbound (free) Metabolite2 molecules in lysosomal space on a single tumor cell
 d/dt(Meta2_cell_lyso_f) <- - K_Meta_lyso_out * (V_cell / V_cell_lyso) * Meta2_cell_lyso_f 
                             + K_Meta_lyso_in * Meta2_cell_cyto_f 
                             - log(2) / DT_tumor * Meta2_cell_lyso_f
                             + K_Drug_cell_met * Drug_cell_lyso_f

 # Number of unbound (free) drug molecules in cytosol on a single tumor cell
 d/dt(Drug_cell_cyto_f) <- K_Drug_lyso_out * (V_cell / V_cell_lyso) * Drug_cell_lyso_f 
                           - K_Drug_lyso_in * Drug_cell_cyto_f 
                           - K_Drug_ex_out * Drug_cell_cyto_f 
                           - K_Drug_cyto_dt_on * SF / V_cell * Drug_cell_cyto_f * (DrugTarget_cell_cyto_t * V_cell / SF - Drug_cell_cyto_b_dt - Meta2_cell_cyto_b_dt) 
                           - K_Drug_met * Drug_cell_cyto_f 
                           + K_Drug_cyto_dt_off * Drug_cell_cyto_b_dt 
                           + K_Drug_ex_in * (V_cell / (V_tumor * E_Drug)) * (Drug_ex_f / SF) 
                           - log(2) / DT_tumor * Drug_cell_cyto_f
                           - K_Drug_cell_met * Drug_cell_cyto_f

 # Number of unbound (free) Metabolite1 molecules in cytosol on a single tumor cell
 d/dt(Meta1_cell_cyto_f) <- K_Meta_lyso_out * (V_cell / V_cell_lyso) * Meta1_cell_lyso_f 
                           - K_Meta_lyso_in * Meta1_cell_cyto_f 
                           - K_Meta_ex_out * Meta1_cell_cyto_f 
                           + K_Meta_ex_in * (V_cell / (V_tumor * E_Meta)) * (Meta1_ex_f / SF) 
                           - log(2) / DT_tumor * Meta1_cell_cyto_f
                           + K_Drug_cell_met * Drug_cell_cyto_f

  # Number of unbound (free) Metabolite2 molecules in cytosol on a single tumor cell
 d/dt(Meta2_cell_cyto_f) <- K_Meta_lyso_out * (V_cell / V_cell_lyso) * Meta2_cell_lyso_f 
                           - K_Meta_lyso_in * Meta2_cell_cyto_f 
                           - K_Meta_ex_out * Meta2_cell_cyto_f 
                           - K_Meta_cyto_dt_on * SF / V_cell * Meta2_cell_cyto_f * (DrugTarget_cell_cyto_t * V_cell / SF - Drug_cell_cyto_b_dt - Meta2_cell_cyto_b_dt) 
                           + K_Meta_cyto_dt_off * Meta2_cell_cyto_b_dt 
                           + K_Meta_ex_in * (V_cell / (V_tumor * E_Meta)) * (Meta2_ex_f / SF) 
                           - log(2) / DT_tumor * Meta2_cell_cyto_f
                           + K_Drug_cell_met * Drug_cell_cyto_f

 # Number of target-bound drug molecules in cytosol on a single tumor cell
 d/dt(Drug_cell_cyto_b_dt) <- K_Drug_cyto_dt_on * SF / V_cell * Drug_cell_cyto_f * (DrugTarget_cell_cyto_t * V_cell / SF - Drug_cell_cyto_b_dt - Meta2_cell_cyto_b_dt) 
                             - K_Drug_cyto_dt_off * Drug_cell_cyto_b_dt 
                             - log(2) / DT_tumor * Drug_cell_cyto_b_dt

 # Number of target-bound Metabolite2 molecules in cytosol on a single tumor cell
 d/dt(Meta2_cell_cyto_b_dt) <- K_Meta_cyto_dt_on * SF / V_cell * Meta2_cell_cyto_f * (DrugTarget_cell_cyto_t * V_cell / SF - Drug_cell_cyto_b_dt - Meta2_cell_cyto_b_dt) 
                             - K_Meta_cyto_dt_off * Meta2_cell_cyto_b_dt 
                             - log(2) / DT_tumor * Meta2_cell_cyto_b_dt

 # Logistic killing function from Thomas Rysiok ------------------------------
 ########################

 if (!is.na(Drug_cell_cyto_f) && Drug_cell_cyto_f > 0) {
   nM_Drug_cell_cyto_f ~ Drug_cell_cyto_f * SF / V_cell
 } else {nM_Drug_cell_cyto_f ~ 0}  # Otherwise log of -1 gets NA

 tl ~ log(nM_Drug_cell_cyto_f) - log(EC50)
 LOGI ~ k_g / (1 + (k_g / k_z - 1) * exp(-k_r * k_g * tl))
 R_Kill ~ k_kill_max * (log(2) / DT_tumor)^f_DT_kill * LOGI

 Kg ~ log(2) / DT_tumor * (1 - V_tumor_pro_mm3 / V_tumor_max) / (1 + (log(2) / DT_tumor * V_tumor_pro_mm3 / k_lin)^psi)^(1/psi)
 
 d/dt(V_tumor_pro_mm3) <- (Kg - R_Kill) * V_tumor_pro_mm3
 d/dt(V_tumor_dyi_1_mm3) <- R_Kill * V_tumor_pro_mm3 - 1 / tau * V_tumor_dyi_1_mm3
 d/dt(V_tumor_dyi_2_mm3) <- 1 / tau * (V_tumor_dyi_1_mm3 - V_tumor_dyi_2_mm3)
 d/dt(V_tumor_dyi_3_mm3) <- 1 / tau * (V_tumor_dyi_2_mm3 - V_tumor_dyi_3_mm3)
